			\item
				Find $\pder{z}{z}$ and $\pder{z}{y}$ for $x^2 - y^2 + z^2 - 2z = 4$
				\begin{itemize}[ ]
					\item \textbf{Solution}
						\begin{align*}
							\pder{}{x}(x^2 - y^2 + z^2 - 2z) &= \pder{}{x}(4) 
								&\pder{}{y}(x^2 - y^2 + z^2 - 2z)
									&= \pder{}{y}(4) \\
							2x - 0 + 2z\pder{z}{x} - 2\pder{z}{x} &= 0
								&0 - 2y + 2z\pder{z}{y} - 2\pder{z}{y}
									&= 0 \\
							\pder{z}{x}(2z - 2) &= -2x
								&\pder{z}{y}(2z - 2)
									&= 2y \\
							\pder{z}{x} &= -\frac{x}{z - 1}
								&\pder{z}{y}
									&= \frac{y}{z - 1}
						\end{align*}
				\end{itemize}
			\item
