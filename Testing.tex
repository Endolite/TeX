\newcommand{\pbet}[5]
{

		\draw[<->] (-3.75, 0) -- (3.75, 0);
			\draw[domain = -3.75:3.75, smooth, variable = \x] plot ({\x}, {(e^(-0.5 * \x * \x))/sqrt(2 * pi)});
			\draw[dashed] (#1, 0) -- (#1, #2);
			\draw[dashed] (#3, 0) -- (#3, #4);
			\fill[lightgray, domain = #1:#3, variable = \x] (#1, 0) -- plot ({\x}, {(e^(-0.5 * \x * \x))/sqrt(2 * pi)}) -- (#3, 0);
			node[below]{x}
	\end{tikzpicture}
}
