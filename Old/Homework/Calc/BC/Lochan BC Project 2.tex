							= \frac{2^{\ln x}\ln(2)}{x}
				\end{align*}

					\dv{x} \frac{e^x - e^{-x}}{e^x + e^{-x}} &\implies u_1 = x, u_2 = -x

							= \frac{e^x + e^{-x}}{e^x - e^x}
				\end{align*}

							= 5^{\sin(x^3)}\qty(3x^2\sin(x^3)\ln(5))
